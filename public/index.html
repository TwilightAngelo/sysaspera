/users (get - все пользователи, post - добавить пользователя)

/users/:id (get - конкретный пользователь, put - изменить пользователя, delete - удалить пользователя)

/users/:id/docs (get - все документы, post - добавить документ)

/users/:id/docs/:id (get - конкретный документ, put - изменить документ, delete - удалить конкретный документы)
